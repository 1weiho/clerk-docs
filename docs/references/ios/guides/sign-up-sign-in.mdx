# Sign Up and Sign In with the iOS SDK

Clerk enables you to build fully custom sign-up and sign-in flows using the [`SignUp`](signup-ref) and [`SignIn`](signin-ref) objects.

## `SignUp`

The [`SignUp`][signup-ref] object contains the state of the current sign-up attempt.

The following example accesses the `SignUp` object to check the current sign-up attempt's status.

```swift
import SwiftUI
import ClerkSDK

struct SignUpStatusView: View {
    @ObservedObject private var clerk = Clerk.shared
    
    var body: some View {
        if let signUp = clerk.client?.signUp {
            Text("The current sign up attempt status is \(signUp.status.rawValue)")
        }
    }
}
```

### Detailed `SignUp` example

The following example demonstrates an approach for creating a custom user signup form using [the email and password](/docs/custom-flows/email-password) strategy. This example assumes that additional requirements, like username or phone, are not enabled. 

```swift
import SwiftUI
import ClerkSDK

struct SignUpView: View {
    @ObservedObject private var clerk = Clerk.shared
    @State private var emailAddress = ""
    @State private var password = ""
    
    var body: some View {
        // if there is already an active SignUp,
        // investigate the SignUp object, to see if
        // the user needs to complete more steps.
        if let signUp = clerk.client?.signUp {
            switch signUp.status {
            case .abandoned:
                //
            case .missingRequirements:
                //
            case .complete:
                //
            }
        } else {
            // if there is not already an active SignUp,
            // collect the information needed to start a new SignUp
            VStack {
                TextField("Email Address", text: $emailAddress)
                SecureField("Password", text: $password)
                Button("Sign Up", action: signUp)
            }
        }
    }
    
    // start the sign up process
    private func signUp() {
        Task {
            do {
                try await clerk.client?.createSignUp(strategy: .standard(emailAddress: emailAddress, password: password))
            } catch {
                print(error.localizedDescription)
            }
        }
    }
}
```

### Detailed `SignIn` example

The following example demonstrates an approach for creating a custom user sign-in form using the [email and password](/docs/custom-flows/email-password) strategy.

```swift
import SwiftUI
import ClerkSDK

struct SignInView: View {
    @ObservedObject private var clerk = Clerk.shared
    @State private var emailAddress = ""
    @State private var password = ""
    
    var body: some View {
        // if there is already an active SignIn,
        // investigate the SignIn object, to see if
        // the user needs to complete more steps.
        if let signIn = clerk.client?.signIn {
            switch signIn.status {
            case .needsIdentifier:
                //
            case .needsFirstFactor:
                //
            case .needsSecondFactor:
                //
            case .needsNewPassword:
                //
            case .complete:
                //
            case .abandoned:
                //
            }
        } else {
            // if there is not already an active SignIn,
            // collect the information needed to start a new SignIn
            VStack {
                TextField("Email", text: $emailAddress)
                SecureField("Password", text: $password)
                Button("Sign In", action: signIn)
            }
        }
    }

    // start the sign in process
    private func signIn() {
        Task {
            do {
                try await clerk.client?.createSignIn(strategy: .identifier(emailAddress, password: password))
            } catch {
                print(error.localizedDescription)
            }
        }
    }
}
```
